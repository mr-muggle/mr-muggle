# 自定义注解实现数据脱敏

现在很多项目中都在后端给前端返回数据时进行脱敏处理，类似的场景很常见，比如说用户的手机号、银行卡号、用户姓名、身份证ID等全部需要用****这类的特殊字符进行代替。 最常见的做法比如说字符替换，使用正则表达式都可以完成。除此以外，还可以自定义注解实现数据脱敏。

# 自定义注解的优点

在于对脱敏业务与序列化处理进行解耦，便于拓展和后期维护。

# 数据脱敏的具体实现

# 自定义jackson数据脱敏注解


/**
 * @description: 自定义脱敏注解
 * @version: 1.0
 */
@Target(ElementType.FIELD)
@Retention(RetentionPolicy.RUNTIME)
@JacksonAnnotationsInside
@JsonSerialize(using = DesensitizationSerializer.class)
public @interface DesensitizationAnnotation {
    DesensitizationStrategy strategy();
}

# 定义数据脱敏策略的枚举类


/**
 * @description: 脱敏策略
 * @version: 1.0
 */
public enum DesensitizationStrategy {
    /**
     * 用户名
     */
    USER_NAME(s -> s.replaceAll("(\\S)\\S(\\S*)", "$1*$2")),
    /**
     * 身份证
     */
    ID_CARD(s -> s.replaceAll("(\\d{4})\\d{10}(\\w{4})", "$1****$2")),
    /**
     * 手机号
     */
    MOBILE_PHONE(s -> s.replaceAll("(\\d{3})\\d{4}(\\d{4})", "$1****$2")),
    /**
     * 地址
     */
    ADDRESS(s -> s.replaceAll("(\\S{4})\\S{2}(\\S*)\\S{2}", "$1****"));


    private final Function<String, String> desensitizer;

    DesensitizationStrategy(Function<String, String> desensitizer) {
        this.desensitizer = desensitizer;
    }

    public Function<String, String> desensitizer() {
        return desensitizer;
    }
}

# 自定义JSON序列化处理器


/**
 * @description: 自定义脱敏注解序列化处理器 用于对使用脱敏注解的字段进行数据脱敏
 * @version: 1.0
 */
public class DesensitizationSerializer extends JsonSerializer<String> implements ContextualSerializer {

    private DesensitizationStrategy strategy;

    @Override
    public void serialize(String s, JsonGenerator jsonGenerator, SerializerProvider serializerProvider) throws IOException {
        jsonGenerator.writeString(strategy.desensitizer().apply(s));
    }

    @Override
    public JsonSerializer<?> createContextual(SerializerProvider serializerProvider, BeanProperty beanProperty) throws JsonMappingException {
        //获取属性上的注解属性
        DesensitizationAnnotation annotation = beanProperty.getAnnotation(DesensitizationAnnotation.class);
        if (Objects.nonNull(annotation) && Objects.equals(String.class, beanProperty.getType().getRawClass())) {
            // 主要代码在这里，获取脱敏的规则
            this.strategy = annotation.strategy();
            return this;
        }
        return serializerProvider.findValueSerializer(beanProperty.getType(), beanProperty);
    }
}

# 定义一个Student实体类，对其中字段进行脱敏处理


@Data
public class StudentPO {
    /**
     * 学生编号
     */
    private String stuCode;
    /**
     * 学生年龄
     */
    private Integer age;
    /**
     * 学生班级
     */
    private String classCode;
    /**
     * 学生姓名
     */
    @DesensitizationAnnotation(strategy = DesensitizationStrategy.USER_NAME)
    private String name;

    /**
     * 学生地址
     */
    @DesensitizationAnnotation(strategy = DesensitizationStrategy.ADDRESS)
    private String address;

    /**
     * 学生电话号码
     */
    @DesensitizationAnnotation(strategy = DesensitizationStrategy.MOBILE_PHONE)
    private String mobilePhone;

    /**
     * 学生身份证号码
     */
    @DesensitizationAnnotation(strategy = DesensitizationStrategy.ID_CARD)
    private String idCard;
}
